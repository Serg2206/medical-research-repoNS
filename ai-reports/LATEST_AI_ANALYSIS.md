```python
#!/usr/bin/env python3
"""Fix CSV file by removing invalid header rows"""

import csv
import os
import sys

input_file = "training-data/medical_training_data.csv"
output_file = "training-data/medical_training_data_temp.csv"

try:
    with open(input_file, "r", encoding="utf-8") as f_in, open(output_file, "w", encoding="utf-8") as f_out:
        reader = csv.reader(f_in)
        writer = csv.writer(f_out)

        is_header = True
        valid_lines = 0
        invalid_lines = 0

        for line in reader:
            if is_header or "," in line:
                writer.writerow(line)
                valid_lines += 1
            else:
                print(f"Removing invalid line {valid_lines+invalid_lines+1}: {','.join(line)}")
                invalid_lines += 1

            is_header = False

    print(f"\nDone! Kept {valid_lines} valid lines, removed {invalid_lines} invalid lines")

    # If everything went well, replace the original file with the new one
    os.replace(output_file, input_file)

except Exception as e:
    print(f"An error occurred: {e}")
```

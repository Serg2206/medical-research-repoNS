```python
#!/usr/bin/env python3
"""Fix CSV file by removing invalid header rows"""

import csv
import sys
import os
import argparse

def parse_args():
    parser = argparse.ArgumentParser(description='Fix CSV file by removing invalid header rows.')
    parser.add_argument('csv_file', help='Path to the CSV file to be fixed.')
    return parser.parse_args()

def main():
    args = parse_args()
    temp_file = args.csv_file + '.tmp'

    with open(args.csv_file, "r", encoding="utf-8") as f_in, open(temp_file, "w", encoding="utf-8") as f_out:
        is_first_line = True
        valid_lines = 0
        invalid_lines = 0

        for line in f_in:
            if is_first_line or "," in line:
                f_out.write(line)
                valid_lines += 1
                is_first_line = False
            else:
                print(f"Removing invalid line {valid_lines + invalid_lines + 1}: {line.strip()}")
                invalid_lines += 1

    os.replace(temp_file, args.csv_file)

    print(
        f"\nDone! Kept {valid_lines} valid lines, removed {invalid_lines} invalid lines"
    )

if __name__ == "__main__":
    main()
```

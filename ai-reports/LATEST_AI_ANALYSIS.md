```python
#!/usr/bin/env python3
"""Fix CSV file by removing invalid header rows"""

import csv
import sys
import os

def fix_csv_headers(file_path):
    # Temporary file to store valid lines
    temp_file_path = file_path + '.tmp'

    with open(file_path, "r", encoding="utf-8") as f, open(temp_file_path, "w", encoding="utf-8") as temp_f:
        is_header = True
        valid_lines = 0
        total_lines = 0

        for line in f:
            total_lines += 1
            if is_header or "," in line:  # Keep the header row and lines with CSV data
                temp_f.write(line)
                valid_lines += 1
                is_header = False
            else:
                print(f"Removing invalid line {total_lines}: {line.strip()}")

    # Rename the temp file
    os.rename(temp_file_path, file_path)

    print(
        f"\nDone! Kept {valid_lines} valid lines, removed {total_lines - valid_lines} invalid lines"
    )

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fix_csv_headers.py <file_path>")
        sys.exit(1)

    fix_csv_headers(sys.argv[1])
```

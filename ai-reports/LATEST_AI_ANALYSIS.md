```python
#!/usr/bin/env python3
"""Fix CSV file by removing invalid header rows"""

import csv
import sys
import os
import tempfile

def is_valid_line(index, line):
    """Check if the line is a valid CSV row"""
    return index == 0 or "," in line

# Create a temporary file
with tempfile.NamedTemporaryFile(mode='w', delete=False, encoding="utf-8") as temp:

    # Open the CSV file for reading
    with open("training-data/medical_training_data.csv", "r", encoding="utf-8") as f:
        reader = csv.reader(f)
        writer = csv.writer(temp)

        valid_lines = 0
        total_lines = 0

        for i, line in enumerate(reader):
            total_lines += 1
            if is_valid_line(i, line):
                writer.writerow(line)
                valid_lines += 1
            else:
                print(f"Removing invalid line {i+1}: {','.join(line)}")

    print(
        f"\nDone! Kept {valid_lines} valid lines, removed {total_lines - valid_lines} invalid lines"
    )

# Replace the original file with the cleaned one
os.replace(temp.name, "training-data/medical_training_data.csv")
```

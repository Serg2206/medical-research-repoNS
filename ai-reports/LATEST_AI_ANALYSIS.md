"""Fix CSV file by removing invalid header rows"""

import csv
import sys
import os
import argparse

def parse_args():
    parser = argparse.ArgumentParser(description='Fix CSV file by removing invalid header rows')
    parser.add_argument('csv_file', help='Path to the CSV file to fix')

def main():
    args = parse_args()

    # Use a temporary file to avoid corrupting the original file
    with open(args.csv_file, "r", encoding="utf-8") as f, \
         open(args.csv_file + ".tmp", "w", encoding="utf-8") as f_tmp:

        valid_lines = 0
        total_lines = 0

        for i, line in enumerate(f):
            total_lines += 1
            # Skip if line is just a header without commas
            if i == 0 or "," in line:  # Keep the header row and lines with CSV data
                f_tmp.write(line)
                valid_lines += 1
            else:
                print(f"Removing invalid line {i+1}: {line.strip()}")

    # Replace the original file with the temporary file
    os.rename(args.csv_file + ".tmp", args.csv_file)

    print(
        f"\nDone! Kept {valid_lines} valid lines, removed {total_lines - valid_lines} invalid lines"
    )

if __name__ == "__main__":
    main()

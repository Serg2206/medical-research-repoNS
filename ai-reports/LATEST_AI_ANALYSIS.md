"""Fix CSV file by removing invalid header rows"""

import csv
import os
import tempfile

# Open the CSV file
with open("training-data/medical_training_data.csv", "r", encoding="utf-8") as f, \
     tempfile.NamedTemporaryFile(mode="w", delete=False) as tmp_file:

    reader = csv.reader(f)
    writer = csv.writer(tmp_file)

    valid_lines = 0
    invalid_lines = 0

    for i, row in enumerate(reader):
        if i == 0 or len(row) > 1:  # Keep the header row and rows with CSV data
            writer.writerow(row)
            valid_lines += 1
        else:
            print(f"Removing invalid line {i+1}: {','.join(row)}")
            invalid_lines += 1

os.rename(tmp_file.name, "training-data/medical_training_data.csv")

print(
    f"\nDone! Kept {valid_lines} valid lines, removed {invalid_lines} invalid lines"

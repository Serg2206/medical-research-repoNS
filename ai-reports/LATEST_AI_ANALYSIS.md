            print(f"Removing invalid line {i+1}: {line.strip()}")

def fix_csv_headers(input_filename):
    """Read a CSV file, remove invalid lines, and write it back."""
    with open(input_filename, "r", encoding="utf-8") as input_file:
        with tempfile.NamedTemporaryFile("w", delete=False, encoding="utf-8") as output_file:
            for line in filter_lines(input_file):
                output_file.write(line)
    
    os.rename(output_file.name, input_filename)

def main():
    fix_csv_headers("training-data/medical_training_data.csv")

if __name__ == "__main__":
    main()

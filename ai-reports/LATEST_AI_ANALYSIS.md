```python
#!/usr/bin/env python3
"""Fix CSV file by removing invalid header rows"""

import csv
import sys

def clean_csv(input_file, output_file):
    with open(input_file, "r", encoding="utf-8") as f_in, open(output_file, "w", encoding="utf-8") as f_out:
        reader = csv.reader(f_in)
        writer = csv.writer(f_out)

        invalid_lines = 0
        for i, row in enumerate(reader):
            if i == 0 or len(row) > 1:  # Keep the header row and rows with CSV data
                writer.writerow(row)
            else:
                invalid_lines += 1
                print(f"Removing invalid line {i+1}: {','.join(row)}")

        print(f"\nDone! Kept {i+1-invalid_lines} valid lines, removed {invalid_lines} invalid lines")

if __name__ == "__main__":
    clean_csv("training-data/medical_training_data.csv", "training-data/cleaned_medical_training_data.csv")
```
